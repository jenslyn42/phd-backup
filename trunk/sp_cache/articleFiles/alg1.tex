
\begin{algorithm}[H!bt]
\dontprintsemicolon
\SetVline

\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}\SetKw{Return}{return}

\KwData
{

	$\mathbf{G(V,E)}$: Graph representation of Map \;
	\textbf{H}: a Max-Heap. Contains the utility score of all possible \spaths in $\mathbf{G(V,E)}$. The \spath is the value $v$ and the utility is the key $k$, $(k, v)$, which the is sorted on.\;
	$\mathbf{\Psi}$: The cache \;
	$\Psi_{size}$: Cache size \;
}

\While{$\Psi.spaceLeft > 0$}
{
	$\spath_{ms} \leftarrow pop(\spath_{as}.max)$ \tcp{pop highest scoring element} \; 
	\If{$\Psi.spaceLeft \leq \spath_{ms}.size$}
	{
		$\Psi.insert(\spath:{ms})$\;
	}
}

\tcp{\emph{Initialize \textbf{H}}}
\textbf{Score}($\chi, \Psi$) \;

\tcp{\emph{Fill cache}}
\While{$| \Psi | \leq \Psi_{size}$ OR $| \textbf{H} | = 0$}
{
	$\spath_{ms}$ = \textbf{H}.pop() \; 
	\If{$\Psi_{size} - | \Psi | \geq | \spath_{ms} |$}
	{
		$\Psi.push(\spath_{ms}.v)$\;
		\textbf{Score}($\chi, \Psi$) \;
		
	}
}

\end{algorithm}
