\begin{frame}[red]
\frametitle{Limitation and solutions of this approach}
Limitations of this approach:
\begin{enumerate}
 \item It is inefficient for big number of users
 \item A proximity detection distance is fixed $(2d\sqrt{2})$
\end{enumerate}
Solutions for problems:
\begin{enumerate}
 \item Grouping of friends
 \item Dynamic grid approach
\end{enumerate}
\end{frame}

\subsection{Solutions for Proximity Detection Approach Problems}
\begin{frame}[red]
\frametitle{Grouping of friends}
\only<1>{
	The solution:
	\begin{itemize}
	\item Users are grouped into friend-groups
	\item A single grid is assigned for every friend-group
	\end{itemize}
	Consequences:
	\begin{itemize}
	\item When user location changes, less secret values must be delivered to LBS server
	\end{itemize}
}

\begin{frame}[red]
\frametitle{Dynamic Grid Approach}
\only<1>{
	The solution:
	\begin{itemize}
	\item Assign a list of grids with decreasing cell size for every friend-group
	\item Extend a basic proximity detection approach to support list of grids
	\end{itemize}
	Consequences:
	\begin{itemize}
	\item Any pair of friends in friend-group will be able to choose a preferred proximity distance from a list
	\end{itemize}
}
